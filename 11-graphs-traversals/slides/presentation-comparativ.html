size_t hash_djb2(const char *str, 
                  size_t table_size) {
    unsigned long hash = 5381;
    int c;
    
    while ((c = *str++)) {
        // hash * 33 + c
        hash = ((hash << 5) + hash) + c;
    }
    
    return hash % table_size;
}

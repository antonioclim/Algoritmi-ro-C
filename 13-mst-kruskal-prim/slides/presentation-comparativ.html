void dijkstra(Graph* g, int src, 
              int* dist, int* prev) {
    MinHeap* h = heap_create(g->V);
    
    for (int i = 0; i < g->V; i++) {
        dist[i] = INT_MAX;
        prev[i] = -1;

        int u = min.vertex;

            int v = e->dest;

            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                prev[v] = u;
                heap_decrease_key(h, v, dist[v]);
            }
        }
    }
    heap_destroy(h);
}

import heapq

def dijkstra(graph, source):
    dist = {v: float('inf') for v in graph}
    prev = {v: None for v in graph}
    dist[source] = 0
    
    pq = [(0, source)]
    visited = set()
    
    while pq:
        d, u = heapq.heappop(pq)
        
        if u in visited:
            continue  # lazy deletion
        visited.add(u)
        
        for v, w in graph[u].items():
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                prev[v] = u
                heapq.heappush(pq, (dist[v], v))
    
    return dist, prev

bool bellman_ford(int V, int E, 
                  Edge edges[], int src,
                  int dist[]) {
    // Ini»õializare
    for (int i = 0; i < V; i++)
        dist[i] = INT_MAX;
    dist[src] = 0;
    
    // V-1 relaxƒÉri
    for (int i = 0; i < V - 1; i++) {

            int u = edges[j].src;
            int v = edges[j].dest;

                dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
            }
        }

        if (dist[edges[j].src] + 
            edges[j].weight < 
            dist[edges[j].dest])
            return false;
    }
    return true;
}

from collections import deque

def mark_negative_cycles(graph, dist):
    affected = set()
    
    # GƒÉsim nodurile afectate
    for u in graph:
        for v, w in graph[u].items():
            if dist[u] + w < dist[v]:
                affected.add(v)

                queue.append(neighbor)
    
    return dist

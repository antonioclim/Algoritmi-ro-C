BloomFilter* bloom_create(size_t n, double fpr) {
    BloomFilter *bf = malloc(sizeof(BloomFilter));
    if (!bf) return NULL;
    
    // Calcul parametri optimali
    bf->size_bits = (size_t)(-n * log(fpr) / 
                    (log(2) * log(2)));
    bf->size_bytes = (bf->size_bits + 7) / 8;
    bf->num_hashes = (size_t)((bf->size_bits / 
                     (double)n) * log(2));
    
    // Alocare »ôi ini»õializare cu 0
    bf->bitmap = calloc(bf->size_bytes, 1);
    if (!bf->bitmap) {
        free(bf);
        return NULL;
    }
    
    bf->count = 0;
    return bf;
}

void bloom_free(BloomFilter *bf) {
    if (bf) {
        free(bf->bitmap);
        free(bf);
    }

4. HyperLogLog ‚Äî Cardinalitate

‚öôÔ∏è C ‚Äî Leading Zeros Manual 
typedef struct {
    uint8_t  *registers;
    size_t   num_registers;  // M = 2^p
    uint8_t  precision;      // p
    double   alpha;
} HyperLogLog;

// NumƒÉrƒÉ leading zeros √Æn hash
static uint8_t count_leading_zeros(uint64_t x, uint8_t max) {
    uint8_t count = 1;
    while (count <= max && ((x >> (max - count)) & 1) == 0) {
        count++;
    }
    return count;
}

void hll_add(HyperLogLog *hll, const char *key) {
    uint64_t hash = hash64(key);
    
    // Primii p bi»õi = index bucket
    size_t bucket = hash >> (64 - hll->precision);
    
    // Restul bi»õilor pentru leading zeros
    uint64_t w = hash << hll->precision;
    uint8_t zeros = count_leading_zeros(w, 64 - hll->precision);
    
    // PƒÉstreazƒÉ maximul
    if (zeros > hll->registers[bucket]) {
        hll->registers[bucket] = zeros;
    }
}

double hll_count(HyperLogLog *hll) {
    double sum = 0.0;
    
    for (size_t j = 0; j < hll->num_registers; j++) {
        sum += pow(2.0, -(double)hll->registers[j]);
    }
    
    double E = hll->alpha * hll->num_registers * 
               hll->num_registers / sum;
    
    // Corec»õii pentru extreme...
    return E;
}

import numpy as np
import mmh3

class SimpleHLL:
    def __init__(self, precision=14):
        self.p = precision
        self.m = 1 << precision  # 2^p
        self.registers = np.zeros(self.m, dtype=np.uint8)
        self.alpha = self._get_alpha()
    
    def _get_alpha(self):
        if self.m == 16: return 0.673
        if self.m == 32: return 0.697
        if self.m == 64: return 0.709
        return 0.7213 / (1 + 1.079/self.m)
    
    def _leading_zeros(self, x, bits):
        # Python: bin() »ôi string ops!
        b = bin(x)[2:].zfill(bits)
        return len(b) - len(b.lstrip('0')) + 1
    
    def add(self, item):
        h = mmh3.hash64(item)[0] & 0xFFFFFFFFFFFFFFFF
        bucket = h >> (64 - self.p)
        w = h << self.p
        zeros = self._leading_zeros(w, 64 - self.p)
        self.registers[bucket] = max(
            self.registers[bucket], zeros
        )
    
    def __len__(self):
        # NumPy vectorizat!
        indicator = np.power(2.0, -self.registers.astype(float))
        return int(self.alpha * self.m**2 / indicator.sum())

typedef struct SkipNode {
    int    key;
    void   *value;
    int    level;
    struct SkipNode **forward;  // array de pointeri
} SkipNode;

SkipNode* create_node(int level, int key, void *val) {
    SkipNode *node = malloc(sizeof(SkipNode));
    node->key = key;
    node->value = val;
    node->level = level;
    
    // Alocare array de pointeri forward
    node->forward = malloc((level + 1) * sizeof(SkipNode*));
    for (int i = 0; i <= level; i++) {
        node->forward[i] = NULL;
    }
    
    return node;
}

SkipNode* search(SkipList *sl, int key) {
    SkipNode *curr = sl->header;
    
    // Traversare de la nivel maxim √Æn jos
    for (int i = sl->level; i >= 0; i--) {

               curr->forward[i]->key < key) {
            curr = curr->forward[i];
        }
    }
    
    curr = curr->forward[0];
    if (curr && curr->key == key) {
        return curr;
    }
    return NULL;
}

import random

class SkipNode:
    def __init__(self, key, value, level):
        self.key = key
        self.value = value
        # Lista Python - dimensiune dinamicƒÉ!
        self.forward = [None] * (level + 1)

class SkipList:
    def __init__(self, max_level=16, p=0.5):
        self.max_level = max_level
        self.p = p
        self.level = 0
        self.header = SkipNode(None, None, max_level)
    
    def _random_level(self):
        lvl = 0
        while random.random() < self.p and lvl < self.max_level:
            lvl += 1
        return lvl
    
    def search(self, key):
        curr = self.header
        
        # Traversare
        for i in range(self.level, -1, -1):
            while (curr.forward[i] and 
                   curr.forward[i].key < key):
                curr = curr.forward[i]
        
        curr = curr.forward[0]
        if curr and curr.key == key:
            return curr.value
        return None
    
    def insert(self, key, value):
        update = [None] * (self.max_level + 1)
        curr = self.header
        
        for i in range(self.level, -1, -1):
            while curr.forward[i] and curr.forward[i].key < key:
                curr = curr.forward[i]
            update[i] = curr
        
        lvl = self._random_level()
        node = SkipNode(key, value, lvl)
        
        for i in range(lvl + 1):
            node.forward[i] = update[i].forward[i]
            update[i].forward[i] = node

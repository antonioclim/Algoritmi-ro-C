def check_brackets(expr):
    stack = []
    pairs = {')':'(', ']':'[', '}':'{'}
    
    for char in expr:
        if char in '([{':
            stack.append(char)
        elif char in ')]}':

            if stack.pop() != pairs[char]:
                return False
    
    return len(stack) == 0

int eval_postfix(char *expr) {
    Stack s;
    stack_init(&s);
    
    for (int i = 0; expr[i]; i++) {
        char c = expr[i];
        if (isdigit(c)) {
            push(&s, c - '0');
        } else if (strchr("+-*/", c)) {
            int b, a;
            pop(&s, &b);
            pop(&s, &a);
            int r;
            switch(c) {
                case '+': r = a + b; break;
                case '-': r = a - b; break;
                case '*': r = a * b; break;
                case '/': r = a / b; break;

            push(&s, r);
        }
    }
    int result;
    pop(&s, &result);
    return result;
}

def eval_postfix(expr):
    stack = []
    ops = {
        '+': lambda a,b: a+b,
        '-': lambda a,b: a-b,
        '*': lambda a,b: a*b,
        '/': lambda a,b: a//b
    }
    
    for token in expr.split():
        if token.lstrip('-').isdigit():
            stack.append(int(token))
        elif token in ops:
            b = stack.pop()
            a = stack.pop()
            stack.append(ops[token](a, b))
    
    return stack.pop()
